function domain_create() {
    local domain=$1
    local description=$2
    if ! openstack domain show $domain; then
        openstack domain create --enable --description "$description" $domain
    fi
}

group_create <domain> <group> <description>

function group_create() {
    local domain=$1
    local group=$2
    local description=$3
    if [[ -n $domain ]]; then
        domain="--domain $domain"
    fi
    if ! openstack group show $group; then
        openstack group create $domain --description "$description" $group
    fi
}

project_create <domain> <project> <description>

function project_create() {
    local domain=$1
    local project=$2
    local description=$3
    if [[ -n $domain ]]; then
        domain="--domain $domain"
    fi
    if ! openstack project show $project; then
        openstack project create $domain --enable --description "$description" $project
    fi
}

user_create <domain> <project> <user>

function user_create() {
    local domain=$1
    local project=$2
    local user=$3
    if [[ -n $domain ]]; then
        domain="--domain $domain"
    fi
    if ! openstack user show $user; then
        openstack user create $domain --project $project --password 0penstack $user
    fi
}
